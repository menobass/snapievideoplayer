1. Basic HTML Embedding

Standard Horizontal Layout

<!-- For horizontal videos or mixed content feeds -->
<div class="video-container standard">
    <iframe 
        src="https://play.3speak.tv/embed?v=owner/permlink&mode=iframe"
        width="100%" 
        height="300"
        style="border: none; border-radius: 8px;"
        allowfullscreen>
    </iframe>
</div>

Vertical/Portrait Layout

<!-- For vertical videos or mobile-first feeds -->
<div class="video-container vertical">
    <iframe 
        src="https://play.3speak.tv/embed?v=owner/permlink&mode=iframe"
        width="100%" 
        height="600"
        style="border: none; border-radius: 8px;"
        allowfullscreen>
    </iframe>
</div>

/* Standard container for horizontal/mixed content */

.video-container.vertical {
    max-width: 400px;
    margin: 20px auto;

/* Mobile-responsive containers */
@media (max-width: 768px) {
    .video-container.vertical {
        max-width: 100%;
    }
    
    .video-container.standard {
        max-width: 100%;
    }
}

3. JavaScript Dynamic Detection

// Example: Detect video orientation via API and adjust container
async function embedVideo(videoId, containerId) {
    try {
        // Fetch video metadata
        const response = await fetch(`/api/embed?v=${videoId}`);
        const videoData = await response.json();
        
        // Get video dimensions (you could add this to your API response)
        const isVertical = videoData.isVertical || false; // Add this field
        
        const container = document.getElementById(containerId);
        const iframe = document.createElement('iframe');
        
        // Adjust container based on video orientation
        if (isVertical) {
            container.className = 'video-container vertical';
            iframe.height = '600';
        } else {
            container.className = 'video-container standard'; 
            iframe.height = '300';
        }
        
        iframe.src = `https://play.3speak.tv/embed?v=${videoId}&mode=iframe`;
        iframe.width = '100%';
        iframe.style.border = 'none';
        iframe.allowFullscreen = true;
        
        container.appendChild(iframe);
        
    } catch (error) {
        console.error('Error embedding video:', error);
    }
}

// Usage
embedVideo('eddiespinod/xpztd24i', 'my-video-container');

4. React Component Example

import React from 'react';

const ThreeSpeakPlayer = ({ 
    videoId, 
    layout = 'standard', // 'standard' | 'vertical'
    className = '' 
}) => {
    const containerClass = `video-container ${layout} ${className}`;
    const height = layout === 'vertical' ? '600' : '300';
    
    return (
        <div className={containerClass}>
            <iframe
                src={`https://play.3speak.tv/embed?v=${videoId}&mode=iframe`}
                width="100%"
                height={height}
                style={{ border: 'none', borderRadius: '8px' }}
                allowFullScreen
                title={`3speak video ${videoId}`}
            />
        </div>
    );
};

// Usage examples
<ThreeSpeakPlayer videoId="eddiespinod/xpztd24i" layout="vertical" />
<ThreeSpeakPlayer videoId="other/horizontal" layout="standard" />

Important: The 3speak player will always maintain its internal 16:9 aspect ratio. Vertical videos will show with black letterboxing on the sides. This is intentional and follows industry standards for predictable layouts.

üöÄ Best Practices

1. Container Sizing Strategy

  • Standard feeds: Use 16:9 containers (like YouTube/Twitter)
  • Mobile-first: Use taller containers to better accommodate vertical content
  • Shorts/Stories: Use 9:16 containers for dedicated vertical feeds
  • Mixed content: Stick with 16:9 for consistency

2. Responsive Design

  • Always use width="100%" for responsive scaling
  • Set appropriate max-widths to prevent oversized players
  • Use CSS aspect-ratio property for modern browsers
  • Test on both portrait and landscape mobile orientations

3. Performance Considerations

  • Use &mode=iframe parameter for minimal UI in embeds
  • Consider lazy loading for feeds with many videos
  • Preload critical above-the-fold videos
